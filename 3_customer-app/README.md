Instructions:
  
  Build & Run: 
  
     sh buildimage.sh
     
     
  Build:
  
     docker-compose build
    
    
    
  Run:
  
     docker-compose up -d
    
  
  
  To view running containers:
  
     docker-compose ps
    
    
    
  Stop:
  
     docker-compose stop
  
  
  Remove:
  
     docker-compose down
